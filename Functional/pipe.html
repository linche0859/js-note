# pipe 和 compose

# pipe

將多個 Function 進行封裝，接著同步的執行。

# 實作介面

pipe = (...fns) => (x) => fns.reduce((v, f) => f(v), x);
1

# 範例

const getName = (person) => person.name;

const uppercase = (string) => string.toUpperCase();

const get6Characters = (string) => string.substring(0, 6);

const reverse = (string) =>
  string
    .split('')
    .reverse()
    .join('');

  name: 'Buckethead',

const pipe = (...functions) => (value) => {
  return functions.reduce(

# compose

將 pipe 轉換成另一個方向執行，即從右到左的運作。

# 實作介面

const compose = (...fns) => (x) => fns.reduceRight((v, f) => f(v), x);

// 或

const compose = (...fns) => (x) => {
  const list = fns.slice();
  while (list.length) {
    x = list.pop()(x);
  }
  return x;
};

# lazy-evaluation

  1. 與其每一次都計算後將結果再進到下一個循環，這次 defer 延遲所有的運算

  2. 每一次的循環都會返回一個包裹層級更多的函數

  3. reduce 最後結果得到一個函數

    function composed(...args){
        return fnN(...fn2( ( fn1( ...args ) )...)
}

    3

  4. 傳入參數後，最終組合函數再由內到外處理參數

  return String(str)

function deDuplicate(list) {
  return Array.from(new Set(list));
}

function skipShortWords(words) {
  return words.filter((word) => word.length > 4);
}
