# 載入 Bootstrap 的 Sass 資源

  1. 安裝 Bootstrap (opens new window)

  2. gulpfile.js 中加入 includePaths 路徑

    const paths = {
  styles: {
    includePaths: './node_modules/bootstrap/scss',
    outputStyle: 'expand'
  }
};

# 合併來自於 NPM 的 JavaScript 資源

  1. 安裝 Jquery

    npm install jquery
    1

  2. gulpfile.js 中新增外部 JavaScript 的來源路徑

    const paths = {
  vendors: {
    src: [
      './node_modules/jquery/dist/jquery.slim.min.js',
      './node_modules/bootstrap/dist/js/bootstrap.bundle.min.js' // 已包含 popper.js
    ],
    dest: 'dist/scripts/'
  }
};

  return gulp

    .pipe($.concat('vendors.js'))
    .pipe(gulp.dest(paths.scripts.dest));
}
